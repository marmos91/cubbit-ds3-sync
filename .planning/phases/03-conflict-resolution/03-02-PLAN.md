<objective>
Implement conflict detection and conflict copy creation in the File Provider extension.

Purpose: This is the core of Phase 3 -- before any write operation (upload, delete, rename, move), the extension performs a HEAD request to compare the stored ETag against the remote ETag. When a mismatch is detected, the local version is uploaded as a conflict copy with the locked naming pattern, and the system re-fetches the remote version. This prevents silent data loss from concurrent edits across multiple devices.

Output: Modified FileProviderExtension with pre-flight conflict checks in all CRUD methods, ETag extraction in S3Lib, ETag persistence after uploads, and conflict IPC notification posting.
</objective>

**A. modifyItem -- content changes (`.contents` branch, ~line 422):**
Before the existing `Task { do { ... try await s3Lib.putS3Item(...) } }` block that handles content uploads (the `default:` case under `changedFields.contains(.contents)`), insert conflict detection inside the Task block, BEFORE the putS3Item call:

```swift
// --- Conflict detection (pre-flight HEAD check) ---
// Skip folders (S3 key prefixes, no content conflicts)
if s3Item.contentType != .folder {
    do {
        let remoteItem = try await s3Lib.remoteS3Item(for: s3Item.itemIdentifier, drive: drive)
        let remoteETag = ETagUtils.normalize(remoteItem.metadata.etag)
        let storedETag = try? await self.metadataStore?.fetchItemEtag(
            byKey: s3Item.itemIdentifier.rawValue, driveId: drive.id
        )
        let normalizedStored = ETagUtils.normalize(storedETag ?? nil)

        if let remoteETag, let normalizedStored, !ETagUtils.areEqual(remoteETag, normalizedStored) {
            // Conflict detected -- upload local version as conflict copy
            self.logger.warning("Conflict detected for \(s3Item.itemIdentifier.rawValue, privacy: .public): remote ETag \(remoteETag, privacy: .public) != stored \(normalizedStored, privacy: .public)")

            let conflictKey = ConflictNaming.conflictKey(
                originalKey: s3Item.itemIdentifier.rawValue,
                hostname: self.hostname
            )

            let conflictS3Item = S3Item(
                identifier: NSFileProviderItemIdentifier(conflictKey),
                drive: drive,
                objectMetadata: s3Item.metadata
            )

            // Upload local version as conflict copy with retry
            let conflictETag = try await withRetries(retries: 3) {
                try await s3Lib.putS3Item(conflictS3Item, fileURL: contents, withProgress: putProgress)
            }

            // Record conflict copy in MetadataStore
            try? await self.metadataStore?.upsertItem(
                s3Key: conflictKey,
                driveId: drive.id,
                etag: ETagUtils.normalize(conflictETag),
                lastModified: Date(),
                syncStatus: .conflict,
                parentKey: s3Item.parentItemIdentifier == .rootContainer ? nil : s3Item.parentItemIdentifier.rawValue,
                size: Int64(documentSize)
            )

            // Send IPC notification for conflict
            nm.sendConflictNotification(
                filename: s3Item.filename,
                conflictKey: conflictKey,
                driveId: drive.id
            )

            // Signal enumerator to re-fetch remote version for original filename
            putProgress.completedUnitCount = numParts
            nm.sendDriveChangedNotificationWithDebounce(status: .idle)
            self.signalChanges()
            cb.handler(conflictS3Item, NSFileProviderItemFields(), false, nil)
            return
        }
    } catch let s3Error as S3ErrorType {
        // HEAD failed -- return transient error for File Provider retry
        self.logger.error("Conflict check HEAD failed: \(s3Error.errorCode, privacy: .public)")
        nm.sendDriveChangedNotificationWithDebounce(status: .error)
        cb.handler(nil, NSFileProviderItemFields(), false, NSFileProviderError(.serverUnreachable) as NSError)
        return
    } catch {
        // Network error during HEAD -- return transient error
        self.logger.error("Conflict check failed: \(error)")
        nm.sendDriveChangedNotificationWithDebounce(status: .error)
        cb.handler(nil, NSFileProviderItemFields(), false, NSFileProviderError(.serverUnreachable) as NSError)
        return
    }
}
// --- End conflict detection ---
```

**C. deleteItem -- check if remote was modified before deleting (~line 603):**
Replace the existing Task block contents. The full new flow inside `Task { do { ... } }`:

```swift
let s3Item = S3Item(
    identifier: identifier,
    drive: drive,
    objectMetadata: S3Item.Metadata(size: 0)
)

// Skip conflict check for folder keys (ending with "/")
let isFolder = identifier.rawValue.hasSuffix(String(DefaultSettings.S3.delimiter))

if !isFolder {
    // --- Conflict detection: check if remote was modified before deleting ---
    do {
        let remoteItem = try await s3Lib.remoteS3Item(for: identifier, drive: drive)
        let remoteETag = ETagUtils.normalize(remoteItem.metadata.etag)
        let storedETag = try? await self.metadataStore?.fetchItemEtag(
            byKey: identifier.rawValue, driveId: drive.id
        )
        let normalizedStored = ETagUtils.normalize(storedETag ?? nil)

        if let remoteETag, let normalizedStored, !ETagUtils.areEqual(remoteETag, normalizedStored) {
            // Remote was modified since last sync -- cancel delete
            self.logger.warning("Delete cancelled: remote ETag changed for \(identifier.rawValue, privacy: .public)")
            nm.sendDriveChangedNotificationWithDebounce(status: .idle)
            self.signalChanges()
            cb.handler(NSFileProviderError(.cannotSynchronize) as NSError)
            return
        }
    } catch let s3Error as S3ErrorType where s3Error.errorCode == "NoSuchKey" || s3Error.errorCode == "NotFound" {
        // Both sides deleted -- treat as success
        self.logger.debug("File already deleted remotely: \(identifier.rawValue, privacy: .public)")
        try? await self.metadataStore?.deleteItem(byKey: identifier.rawValue, driveId: drive.id)
        progress.completedUnitCount = 1
        nm.sendDriveChangedNotificationWithDebounce(status: .idle)
        self.signalChanges()
        cb.handler(nil)
        return
    } catch {
        // HEAD failed (network) -- return transient error for retry
        self.logger.error("Delete conflict check HEAD failed: \(error)")
        nm.sendDriveChangedNotificationWithDebounce(status: .error)
        cb.handler(NSFileProviderError(.serverUnreachable) as NSError)
        return
    }
    // --- End conflict detection ---
}

nm.sendDriveChangedNotification(status: .sync)
try await s3Lib.deleteS3Item(s3Item, withProgress: progress)
self.logger.debug("S3Item with identifier \(identifier.rawValue, privacy: .public) deleted successfully")
```
